package bomb.enumerations;

import bomb.interfaces.Labeled;

public enum Roots implements Labeled {
    A("A", "+3 +5 +3"), A_SHARP("A#","+4 +4 +3"), B("B","+3 +4 +3"),
    C("C","+3 +3 +4"), C_SHARP("C#","+2 +1 +4"), D("D","+4 +3 +4"),
    D_SHARP("D#","+3 +1 +6"), E("E","+5 +2 +3"), F("F","+2 +2 +3"),
    F_SHARP("F#","+4 +3 +3"), G("G","+3 +4 +4"), G_SHARP("G#","+4 +4 +2");

    private final String note, label;

    @Override
    public String getLabel() {
        return label;
    }

    public String getNote() {
        return note;
    }

    Roots(String note, String label){
        this.note = note;
        this.label = label;
    }
}
